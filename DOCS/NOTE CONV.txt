#include <iostream>
#include <string>

int main()
{
    
    std::string notes[] =
    {
        "nC", "nCs", "nD", "nEb", "nE", "nF", "nFs", "nG", "nGs", "nA", "nBb", "nB" 
    };
    
    int arr[] = 
    {
        0x9f, 0x01, 0x9e, 0x9d, 0x9d, 0x9c, 0x9d, 0x9d, 0x9e, 0x9e, 0x9f, 0xa0, 0x9d, 0x9d,
        0x9c,  0x9b,  0x9b,  0x9a,  0x9b, 0x9c,  0x9c,  0x9d,  0x9d,  0x9e,  0x9c,  0x9b, 
        0x9b,  0x9a,  0x99,  0x98,  0x99,  0x9a,  0x9b,  0x9c,  0x9c,  
        0x9d,  0x9b,  0x9a,  0x99,  0x98,  0x97,  0x97,  0x98,  
        0x99,  0x9a,  0x9a,  0x9b,  0x9c,  0x99,  0x98,  0x97,  
        0x96,  0x95,  0x95,  0x96,  0x97,  0x98,  0x99,  0x9a,  
        0x9a,  0x98,  0x96,  0x95,  0x94,  0x93,  0x94,  0x95,  
        0x96,  0x97,  0x87,  0x93,  0x9a,  0x9f,  0xa3,  0xa6,  
        0xa9,  0xab,  0xad,  0xaf,  0xb1,  0x80,  0x87,  0x93,  
        0x9a,  0x9f,  0xa3,  0xa6,  0xa9,  0xab,  0xad,  0xaf,  
        0xb1
    };
    
    
    for (int i = 0; i < 92; i++)
    {
        if (arr[i] == 0x80)
            std::cout << "nRst, ";
            
        else if (arr[i] > 0x80)
        {
            int val = arr[i] - 0x81;
            val %= 0x0c;
            std::cout << notes[val] << (arr[i] - 0x81 + 12) / 12 << ", ";
        }
        
        if (i % 12 == 0 && i != 0)
            std::cout << "\n";
    }
    return 0;
}
